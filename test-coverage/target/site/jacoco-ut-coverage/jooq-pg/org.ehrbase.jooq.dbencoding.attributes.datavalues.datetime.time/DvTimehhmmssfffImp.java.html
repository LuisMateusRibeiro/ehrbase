package org.ehrbase.jooq.dbencoding.attributes.datavalues.datetime.time;

import com.nedap.archie.rm.datavalues.quantity.datetime.DvTime;
import java.time.temporal.Temporal;
import org.ehrbase.jooq.dbencoding.attributes.datavalues.datetime.TemporalAttributes;

public class DvTimehhmmssfffImp extends DvTimeAttributesImp {

    public DvTimehhmmssfffImp(DvTime dvTime) {
        super(dvTime);
    }

    @Override
    public Temporal getValueExtended() {
        return (Temporal) dvTime.getValue();
    }

    @Override
    public Integer getSupportedChronoFields() {
        return supportedChronoFields(TemporalAttributes.HOUR | MINUTE_OF_HOUR | SECOND_OF_MINUTE | MILLI_OF_SECOND);
    }
}
